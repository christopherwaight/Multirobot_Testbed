1function [Vf] = SwarmSimWeightedAttractNew(RobotParams, NRobot, SensorRange,WeightFunction,SysParam)
 2%SwarmSimWeightedAttract Weighted attract behavior for Swarm_Adaptive_Navigation_Simulator.slx
 3%   LATEST UPDATE: 08/30/2018 by NJM 
 4%   This attract behavior is called used by other composite behaviors in 
 5%   Swarm_Adaptive_Naviagtion_Simulator. The weighted attract behavior
 6%   causes the robots to clump together but favors weighting toward certain
 7%   robots. It is set for constant velocity and for constant force. 
 8
 9%   WeightFunctions take a robot number for "self" and the robot params and
10%   need to return a 1 by N vector of weights
11
12
13% Determine number of robots based off length of robot params vector 
14N= floor(length(RobotParams)/4); 
15
16%% initialize variables 
17% position variables for 3-DOF omnibot are x,y,and theta (rotation about z-axis)
18x = zeros(2*N,1); 
19
20% d - distance between "self" and other robot
21d=zeros(1,2*N); 
22
23%Vft is final angular velocity about z
24Vft=0;
25% Vf - final velocity as a vector.
26Vf=[0.0 0.0 0.0];
27
28
29%% Set x and y inputs into an array
30x(1:2:end) = RobotParams(1:4:end); 
31x(2:2:end) = RobotParams(2:4:end); 
32d(1:2:2*N) = sqrt((x(1:2:end)-x(2*NRobot-1)).^2 + (x(2:2:end)-x(2*NRobot)).^2); 
33d(2:2:2*N) = sqrt((x(1:2:end)-x(2*NRobot-1)).^2 + (x(2:2:end)-x(2*NRobot)).^2); 
34d(d>SensorRange| d==0)=inf;
35P = diag(1./d);
36
37%% Attract Behavior 
38
39% Determine distance and angle to each robot 
40L = eye(2*N);
41L(1:2:end,2*NRobot-1) = L(1:2:2*N,2*NRobot-1)-1;
42L(2:2:end,2*NRobot) = L(2:2:2*N,2*NRobot)-1;
43W = WeightFunction(NRobot,RobotParams,SensorRange, [SysParam]);
44
45% If robot is within sensor range of  Nrobot, it is attracted to that
46% robot. If the  robot is outside of sensor range, then that robot has
47% no effect on NRobot's velocity. 
48
49v = W*P*L*x; 
50%Calculate velocity vector components: 
51Vfx= v(1);
52Vfy= v(2); 
53mag = sqrt(Vfx^2+Vfy^2); 
54
55Vf(1,1:3)= [Vfx Vfy Vft]/mag;
56if isnan(sum(Vf))
57    Vf = [0 0 0];
58end
59end
