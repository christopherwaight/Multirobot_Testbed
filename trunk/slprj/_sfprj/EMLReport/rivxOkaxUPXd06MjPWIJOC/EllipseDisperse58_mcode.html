1function [W] = EllipseDisperse(NRobot,RobotParams,DL,SysParam)
 2N = floor(length(RobotParams)/4);
 3SV = zeros(2,2*N); 
 4% Need three parameters specified: semimajor axis (a), eccentricity (e), 
 5% and angle of ellipse (alpha)
 6a = 10; %m 
 7e =0.9; 
 8alpha = 45; 
 9
10% semiminor axis(b) is specified by eccentricity
11b = a * sqrt(1-e^2);
12
13xs = RobotParams(1:4:end);
14ys = RobotParams(2:4:end); 
15ss = RobotParams(4:4:end);
16ds = zeros(N,1); 
17ds = sqrt((xs-xs(NRobot)).^2 + (ys-ys(NRobot)).^2); 
18xs(ds>DL) = nan; 
19ys(ds>DL) = nan; 
20ds(ds>DL) = nan; 
21
22%center of ellipse is specified by the COM (xc,yc)
23xc = mean(xs,'omitnan');
24yc = mean(ys,'omitnan'); 
25
26EQ = (((xs(NRobot)-xc)*cosd(alpha)+ (ys(NRobot)-yc)*sind(alpha)).^2 )./ (a^2) + (((xs(NRobot)-xc)*sind(alpha)- (ys(NRobot)-yc)*cosd(alpha)).^2 )./ (b^2);
27
28W = zeros(2,2*N); 
29if EQ <=1
30    %     [~,mini] = mink(ds,2);
31    %     w(mini) = -ds(mini);
32else
33    beta = -0.5*e+0.5; 
34    W(1, 1:2:end) = (sind(alpha)^2 + beta*cosd(2*alpha))*ds;
35    W(1, 2:2:end) = cosd(alpha)*sind(alpha)*(2*beta-1)*ds;
36    W(2, 1:2:end) = cosd(alpha)*sind(alpha)*(2*beta-1)*ds;
37    W(2, 2:2:end) = (cosd(alpha)^2 - beta*cosd(2*alpha))*ds;
38end
39
40W(isnan(W)) = 0;
41
42end
